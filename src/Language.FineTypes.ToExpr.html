{-# LANGUAGE FlexibleInstances #-}
{-# OPTIONS_GHC -Wno-orphans #-}

module Language.FineTypes.ToExpr where

import Prelude

import Data.Void (Void)
import Text.Megaparsec
    ( ParseErrorBundle (..)
    )

import Data.TreeDiff (ToExpr (..))
import GHC.IO.Exception (IOException)
import Text.Megaparsec.Error (ParseError (..))

instance ToExpr (ParseErrorBundle String Void) where
    toExpr :: ParseErrorBundle String Void -> Expr
toExpr ParseErrorBundle String Void
_ = String -> Expr
forall a. ToExpr a => a -> Expr
toExpr String
"ParseErrorBundle"

instance ToExpr (ParseError String Void) where
    toExpr :: ParseError String Void -> Expr
toExpr ParseError String Void
_ = String -> Expr
forall a. ToExpr a => a -> Expr
toExpr String
"ParseError"

instance ToExpr IOException where
    toExpr :: IOException -> Expr
toExpr IOException
_ = String -> Expr
forall a. ToExpr a => a -> Expr
toExpr String
"IOException"
